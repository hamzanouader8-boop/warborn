--// IsValidTarget (Only real players)
local function IsValidTarget(plr)
   if plr == LocalPlayer then return false end
   if not plr.Character or not plr.Character:FindFirstChild("Humanoid") or plr.Character.Humanoid.Health <= 0 then return false end
   
   if not getgenv().TeamCheck then return true end
   
   if plr.Team and LocalPlayer.Team then
      return plr.Team ~= LocalPlayer.Team
   end
   
   local f1 = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("Faction")
   local f2 = plr.Character:FindFirstChild("Faction")
   if f1 and f2 then
      return f1.Value ~= f2.Value
   end
   
   return true
end

--// Visibility Check
local function IsVisible(targetPos)
   if not getgenv().VisibilityCheck then return true end
   local origin = Camera.CFrame.Position
   local direction = (targetPos - origin).Unit * 5000
   local params = RaycastParams.new()
   params.FilterType = Enum.RaycastFilterType.Exclude
   params.FilterDescendantsInstances = {LocalPlayer.Character or {}}
   local result = workspace:Raycast(origin, direction, params)
   return not result or (result.Position - targetPos).Magnitude < 5
end

--// Prediction
local function Predict(part, isMobile)
   local predictVal = isMobile and getgenv().MobilePredict or getgenv().PredictValue
   local hrp = part.Parent:FindFirstChild("HumanoidRootPart")
   if getgenv().Prediction and hrp then
      return part.Position + hrp.AssemblyLinearVelocity * predictVal
   end
   return part.Position
end

--// GetTarget PC
local function GetTargetPC()
   local mouse = UserInputService:GetMouseLocation()
   local closest, dist = nil, math.huge
   for _,plr in Players:GetPlayers() do
      if IsValidTarget(plr) and plr.Character:FindFirstChild(getgenv().AimPart) then
         local pred = Predict(plr.Character[getgenv().AimPart], false)
         if IsVisible(pred) then
            local pos, onscreen = Camera:WorldToViewportPoint(pred)
            if onscreen then
               local mag = (Vector2.new(pos.X, pos.Y) - mouse).Magnitude
               if mag < getgenv().FOVRadius and mag < dist then
                  dist = mag; closest = plr
               end
            end
         end
      end
   end
   return closest
end

--// GetTarget Mobile

         if UserInputService:IsKeyDown(Enum.KeyCode.S) then moveDir -= Camera.CFrame.LookVector end
         if UserInputService:IsKeyDown(Enum.KeyCode.A) then moveDir -= Camera.CFrame.RightVector end
         if UserInputService:IsKeyDown(Enum.KeyCode.D) then moveDir += Camera.CFrame.RightVector end
         if UserInputService:IsKeyDown(Enum.KeyCode.Space) then moveDir += Vector3.yAxis end
         if UserInputService:IsKeyDown(Enum.KeyCode.LeftControl) then moveDir -= Vector3.yAxis end
         root.Velocity = moveDir.Magnitude > 0 and (moveDir.Unit * getgenv().FlySpeed) or Vector3.new()
      end
   end
end)

--// ESP Tables
local ESP = {}

local function CreateESP(plr)
   if plr == LocalPlayer or ESP[plr] then return end
   
   local box = Drawing.new("Square")
   box.Thickness = 2
   box.Filled = false
   box.Color = Color3.fromRGB(255,0,0)
   box.Transparency = 0.7
   box.Visible = false

   local skeleton = {}
   for i=1,8 do 
      local line = Drawing.new("Line")
      line.Thickness = 1
      line.Color = Color3.fromRGB(255,255,255)
      line.Visible = false
      skeleton[i] = line 
   end

   local nameT = Drawing.new("Text")
   nameT.Size = 14
   nameT.Color = Color3.fromRGB(255,255,255)
   nameT.Outline = true
   nameT.Visible = false

   local tracer = Drawing.new("Line")
   tracer.Thickness = 2
   tracer.Color = Color3.fromRGB(0,255,255)
   tracer.Visible = false

   local distT = Drawing.new("Text")
   distT.Size = 12
   distT.Color = Color3.fromRGB(255,255,0)
   distT.Outline = true
   distT.Visible = false

   local cham = Instance.new("Highlight")
   cham.FillColor = Color3.fromRGB(255,0,0)
   cham.OutlineColor = Color3.fromRGB(255,255,255)
   cham.FillTransparency = 0.5
   cham.OutlineTransparency = 0
   cham.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
   cham.Enabled = false

   ESP[plr] = {box=box, skeleton=skeleton, name=nameT, tracer=tracer, dist=distT, cham=cham}
end

--// ESP RenderStepped (FIXED BOX)
RunService.RenderStepped:Connect(function()
   local mouseLoc = UserInputService:GetMouseLocation()
   FOV.Position = mouseLoc
   FOV.Radius = getgenv().FOVRadius
   FOV.Visible = getgenv().FOVCircleEnabled

   for plr, esp in pairs(ESP) do
      if IsValidTarget(plr) and plr.Character and plr.Character:FindFirstChild("Head") and plr.Character:FindFirstChild("HumanoidRootPart") then
         local headPos, headVis = Camera:WorldToViewportPoint(plr.Character.Head.Position + Vector3.new(0, 0.4, 0))
         local rootPos, rootVis = Camera:WorldToViewportPoint(plr.Character.HumanoidRootPart.Position)
         local feetPos, _ = Camera:WorldToViewportPoint(plr.Character.HumanoidRootPart.Position - Vector3.new(0, 3, 0))

         if headVis and rootVis then
            -- Better box positioning
            local topY    = headPos.Y - 20    -- margin above head
            local bottomY = feetPos.Y + 15    -- margin below feet
            local height  = bottomY - topY
            local width   = height * 0.58     -- adjust this 0.58 for wider/thinner box

            local centerX = rootPos.X

            esp.box.Size     = Vector2.new(width, height)
            esp.box.Position = Vector2.new(centerX - width/2, topY)
            esp.box.Visible  = getgenv().EspEnabled

            -- Name above
            if getgenv().EspNames then
               esp.name.Text = plr.Name
               esp.name.Position = Vector2.new(centerX, topY - 18)
               esp.name.Visible = true
            else
               esp.name.Visible = false
            end

            -- Tracer from top of screen to top of box
            if getgenv().EspTracers then
               esp.tracer.From = Vector2.new(mouseLoc.X, 0)
               esp.tracer.To   = Vector2.new(centerX, topY)
               esp.tracer.Visible = true
            else
               esp.tracer.Visible = false
            end

            -- Distance below
            if getgenv().EspDistance then
               local dist = (LocalPlayer.Character.HumanoidRootPart.Position - plr.Character.HumanoidRootPart.Position).Magnitude
               esp.dist.Text = math.floor(dist + 0.5) .. "m"
               esp.dist.Position = Vector2.new(centerX, bottomY + 5)
               esp.dist.Visible = true
            else
               esp.dist.Visible = false
            end

            -- Simple skeleton line (head to root)
            if getgenv().EspSkeleton then
               if esp.skeleton[1] then
                  esp.skeleton[1].From = Vector2.new(headPos.X, headPos.Y)
                  esp.skeleton[1].To   = Vector2.new(rootPos.X, rootPos.Y)
                  esp.skeleton[1].Visible = true
               end
            else
               for _, ln in pairs(esp.skeleton) do ln.Visible = false end
            end

            esp.cham.Adornee = plr.Character
            esp.cham.Enabled = getgenv().EspChams
         else
            esp.box.Visible = false
            for _, ln in pairs(esp.skeleton) do ln.Visible = false end
            esp.name.Visible = false
            esp.tracer.Visible = false
            esp.dist.Visible = false
            esp.cham.Enabled = false
         end
      else
         esp.box.Visible = false
         for _, ln in pairs(esp.skeleton) do ln.Visible = false end
         esp.name.Visible = false
         esp.tracer.Visible = false
         esp.dist.Visible = false
         esp.cham.Enabled = false
      end
   end
end)

--// TP Pull & Kill All
local function TPPullEnemies()
   if not LocalPlayer.Character or not LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then 
      return Rayfield:Notify({Title="Error",Content="No Character!",Duration=3}) 
   end
   local myCFrame = LocalPlayer.Character.HumanoidRootPart.CFrame
   local count = 0
   for _,plr in Players:GetPlayers() do
      if IsValidTarget(plr) and plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
         plr.Character.HumanoidRootPart.CFrame = myCFrame * CFrame.new(math.random(-5,5)*0.1, math.random(-2,2)*0.1, math.random(-5,5)*0.1)
         count += 1
      end
   end
   Rayfield:Notify({Title="TP Pull",Content=count.." Enemies Pulled!",Duration=3})
end

local function KillAllEnemies()
   local count = 0
   for _,plr in Players:GetPlayers() do
      if IsValidTarget(plr) and plr.Character and plr.Character:FindFirstChild("Humanoid") then
         plr.Character.Humanoid.Health = 0
         count += 1
      end
   end
   Rayfield:Notify({Title="Kill All",Content=count.." Enemies Killed!",Duration=3})
end
